<details><summary><b> <a href="https://www.acwing.com/problem/content/1017/">摘花生</a></b> </summary>

```cpp
#include<iostream>
using namespace std
const int N = 101;
int main(){
    int t;
    cin >> t;
    while(t--){
        int r,c;
        cin >> r >> c;
        int f[N][N], w[N][N];
        for(int i = 1; i <= r; i ++){
            for(int j = 1; j <= c; j ++){
                cin >> w[i][j];
            }
        }
        for(int i = 1; i <= r; i ++){
            for(int j = 1; j <= c; j ++){
                f[i][j] = max(f[i-1][j],f[i][j-1]) + w[i][j];
            }
        }
        cout << f[r][c] << endl;
    }
}
```

<details><summary><b><a href="https://www.acwing.com/problem/content/1020/"> 最低通行费</a></b> </summary>

```cpp
#include<iostream>
#include <cstring>
using namespace std;
const int N = 101;
const int INF = 1e9;
int main(){
    int n;
    cin >> n;
    int m[N][N];
    int f[N][N]={INF};
    memset(f, 0x3f3f3f3f, sizeof f);
    for(int i = 1; i <= n; i ++){
        for(int j = 1; j <= n; j ++){
            cin >> m[i][j];
        }
    }
    
    for(int i = 1; i <= n; i ++){
        for(int j = 1; j <= n; j ++){
            if(i == 1 && j == 1)
                f[1][1] = m[1][1];
            else
            f[i][j] = min(f[i-1][j],f[i][j-1]) + m[i][j];
        }
    }
    cout << f[n][n];
}
```
